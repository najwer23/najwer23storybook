# Prerequisites

```js
"react": "^18.3.1"
"react-dom": "^18.3.1"
"typescript": "^5.4.5"
"storybook": "^8.0.10"
"vite": "^5.2.11"
```

# Install
```js
yarn add najwer23storybook --dev
```
```js
npm install najwer23storybook --save-dev
```

# Example

```typescript
import { Button } from "najwer23storybook/lib/Button";

export const Example = () => {
    return <Button
        ariaLabel="button-click"
        onClick={() => { }}
        text="Button"
        type="button"
    />
}
```
